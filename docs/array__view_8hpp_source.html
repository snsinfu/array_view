213  /// Compares views for shallow equality.
214  ///
215  /// Two views are equal if and only if the viewed memory region is exactly
216  /// the same. Empty views are the same if the beginning pointers match;
217  /// arbitrary empty views are not necessarily the same. Still,
218  /// default-constructed views are guaranteed to be the same.
219  ///
220  /// This is a *shallow* comprison. Use `std::lexicographic_compare`
221  /// algorithm for deep element-wise comparisons.
222  ///
223  /// @return true if lhs and rhs are the same view, false otherwise.
224  //
225  /// @par Complexity
226  /// O(1)
227  template<typename T>
228  bool operator==(
229  array_view<T> const& lhs, array_view<T> const& rhs) noexcept
230  {
231  return lhs.data() == rhs.data() && lhs.size() == rhs.size();
232  }
233

238  {
239  return lhs == rhs.as_const();

321  /// Creates an array_view of the elements of a contiguous container.
322  ///
323  /// For an array the pointer to the first element and the compile-time
324  /// length of the array are used to create a view. Otherwise `cont.data()`
325  /// is assumed to be the beginning of the memory region owned by the
326  /// container and `cont.size()` is assumed to be the number of elements in
327  /// that memory region. The behavior is undefined if these assumptions are
328  /// not met.
329  ///
330  /// @param cont A contiguous container or an array.
331  ///
332  /// @return A view of all the elements in the given container.
333  ///
334  /// @par Rationale
335  /// Container-to-array_view conversion is unsafe. An array_view can easily
336  /// outlive the viewed container and cause a dangling pointer bug. Such a
337  /// conversion should not be overlooked in the source code. Thus, this
338  /// library disallows implicit conversion and forces explicit conversion
339  /// via this function.
340  template<typename Cont,
341  typename P = decltype(detail::data(std::declval<Cont&>())),
342  typename S = decltype(detail::size(std::declval<Cont&>()))>
343  constexpr array_view<detail::deref_t<P>> view(Cont& cont) noexcept
344  {
345  return {detail::data(cont), detail::size(cont)};
346  }
